typedef enum rating {
	NONE = 0, ONESTAR = 1, TWOSTAR = 2, THREESTAR = 3, FOURSTAR = 4, FIVESTAR = 5
} Rating;

Duration Struct

typedef struct duration {
	int min;
	int sec;
} Duration;

Record Struct

typedef struct record {
	char artist[30];
	char album[30];
	char song[30];
	char genre[15];
	Duration duration;
	int timesPlayed;
	Rating rating;
	unsigned long id;

} Record;

Node Struct

typedef struct node {
	Record record;
	struct node *pNext;
	struct node *pPrev;

} Node;

linked list struct
